    def sierpinski(t, depth, size, scale_factor=1):
        if depth == 1:
            t.circle(size)
        else:
            t.speed('fastest')
            t.pencolor('red')
            sierpinski(t, depth-1, size/2)
            t.pu()
            t.fd(size/2)
            t.pd()
            t.pencolor('black')
            sierpinski(t, depth-1, size/2)
            t.pu()
            t.bk(size/2)
            t.lt(60)
            t.fd(size/2)
            t.pd()
            t.rt(60)
            t.pencolor('cyan')
            sierpinski(t, depth-1, size/2)
            t.rt(120)
            t.pu()
            t.fd(size/2)
            t.lt(120)
            t.pd()
